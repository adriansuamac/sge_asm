### EJERCICIO 8

``` python

import random

puntos_usuario = 0
puntos_ordenador = 0
opciones = ["piedra", "papel", "tijeras", "lagarto", "spock"]

while puntos_usuario < 5 and puntos_ordenador < 5:
    usuario = input("Elige entre piedra, papel, tijeras, lagarto o spock: ").lower()
    if usuario not in opciones:
        print("No hay esa opcion, intenta de nuevo")
        continue

    ordenador = random.choice(opciones)
    print("El ordenador ha elegido:", ordenador)

    match (usuario, ordenador):
        case (u, o) if u == o:
            print("Empate")
        case ("tijeras", "papel") | ("tijeras", "lagarto"):
            print("¡Ganas esta ronda!")
            puntos_usuario += 1
        case ("papel", "piedra") | ("papel", "spock"):
            print("¡Ganas esta ronda!")
            puntos_usuario += 1
        case ("piedra", "lagarto") | ("piedra", "tijeras"):
            print("¡Ganas esta ronda!")
            puntos_usuario += 1
        case ("lagarto", "spock") | ("lagarto", "papel"):
            print("¡Ganas esta ronda!")
            puntos_usuario += 1
        case ("spock", "tijeras") | ("spock", "piedra"):
            print("¡Ganas esta ronda!")
            puntos_usuario += 1
        case _:
            print("El ordenador gana esta ronda")
            puntos_ordenador += 1

    print(f"Marcador -> Tú: {puntos_usuario} | Ordenador: {puntos_ordenador}")

if puntos_usuario == 5:
    print("Has ganado")
else:
    print("El ordenador ha ganado")


```

    print("1")
elif(n == 2):
    print("1, 1" )
else:
    fib1 = 1
    fib2 = 1
    final = "1, 1"
