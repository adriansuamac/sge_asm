### Subo el contenido del archivo docker-compose.yml

```` yml
version: '3.7'

services:
  postgres_dev:
    image: postgres:14
    container_name: db_dev
    environment:
      - POSTGRES_DB=odoo_dev_db
      - POSTGRES_USER=odoo
      - POSTGRES_PASSWORD=paso
    volumes:
      - ~/OdooDev/dataPG:/var/lib/postgresql/data
    restart: always

  odoo_dev:
    image: odoo:16
    container_name: odoo_dev
    environment:
      - HOST=postgres_dev
      - USER=odoo
      - PASSWORD=paso
    ports:
      - '8069:8069'
    volumes:
      - ~/OdooDev/volumesOdoo/addons:/mnt/extra-addons
      - ~/OdooDev/volumesOdoo/filestore:/var/lib/odoo/filestore
    depends_on:
      - postgres_dev
    command: --dev=all
    restart: always

  postgres_prod:
    image: postgres:14
    container_name: db_prod
    environment:
      - POSTGRES_DB=odoo_prod_db
      - POSTGRES_USER=odoo
      - POSTGRES_PASSWORD=paso
    restart: always

  odoo_prod:
    image: odoo:16
    container_name: odoo_prod
    environment:
      - HOST=postgres_prod
      - USER=odoo
      - PASSWORD=paso
    ports:
      - '8070:8069'
    depends_on:
      - postgres_prod
    restart: always

````
